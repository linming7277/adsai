### 1. зҠ¶жҖҒз®ЎзҗҶжһ¶жһ„

#### Zustand Store и®ҫи®Ў

**з”ЁжҲ·зҠ¶жҖҒ (useUserStore.ts)**:
```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface UserState {
  user: User | null;
  token: string | null;
  subscription: Subscription | null;
  tokenBalance: number;

  // Actions
  setUser: (user: User) => void;
  setToken: (token: string) => void;
  updateTokenBalance: (balance: number) => void;
  logout: () => void;
}

export const useUserStore = create<UserState>()(
  persist(
    (set) => ({
      user: null,
      token: null,
      subscription: null,
      tokenBalance: 0,

      setUser: (user) => set({ user }),
      setToken: (token) => set({ token }),
      updateTokenBalance: (balance) => set({ tokenBalance: balance }),
      logout: () => set({ user: null, token: null }),
    }),
    {
      name: 'user-storage',
    }
  )
);
```

**Offer зҠ¶жҖҒ (useOfferStore.ts)**:
```typescript
import { create } from 'zustand';

interface OfferState {
  offers: Offer[];
  selectedOffers: string[];
  filters: OfferFilters;

  // Actions
  setOffers: (offers: Offer[]) => void;
  addOffer: (offer: Offer) => void;
  updateOffer: (id: string, data: Partial<Offer>) => void;
  deleteOffer: (id: string) => void;
  toggleSelectOffer: (id: string) => void;
  setFilters: (filters: OfferFilters) => void;
}

export const useOfferStore = create<OfferState>((set) => ({
  offers: [],
  selectedOffers: [],
  filters: {},

  setOffers: (offers) => set({ offers }),
  addOffer: (offer) => set((state) => ({
    offers: [...state.offers, offer]
  })),

    offers: state.offers.filter((o) => o.id !== id)
  })),
  toggleSelectOffer: (id) => set((state) => ({
    selectedOffers: state.selectedOffers.includes(id)
      ? state.selectedOffers.filter((oid) => oid !== id)
      : [...state.selectedOffers, id]
  })),
  setFilters: (filters) => set({ filters }),
```

```typescript
import { useEffect, useState } from 'react';

export function useCountUp(
  end: number,
  duration: number = 1000
) {
  const [count, setCount] = useState(0);

  useEffect(() => {
    let startTime: number;
    const startValue = 0;

    const animate = (currentTime: number) => {
      if (!startTime) startTime = currentTime;
      const progress = Math.min((currentTime - startTime) / duration, 1);

      setCount(Math.floor(progress * (end - startValue) + startValue));

      if (progress < 1) {
        requestAnimationFrame(animate);
      }
    };

    requestAnimationFrame(animate);
  }, [end, duration]);

  return count;
}
```
