### 2. 前端 OAuth 流程实现

前端需要实现的 OAuth 流程：

```typescript
// lib/google-ads/oauth.ts

const GOOGLE_ADS_CLIENT_ID = process.env.NEXT_PUBLIC_GOOGLE_ADS_CLIENT_ID;
const REDIRECT_URI = process.env.NEXT_PUBLIC_GOOGLE_ADS_OAUTH_REDIRECT_URI;

export function initiateGoogleAdsOAuth() {
  // 1. 调用后端获取 OAuth URL
  const response = await fetch(
    `${API_BASE_URL}/adscenter/oauth/url?redirect_uri=${REDIRECT_URI}`
  );

  const { auth_url, state } = await response.json();

  // 2. 存储 state 到 sessionStorage（防止 CSRF）
  sessionStorage.setItem('oauth_state', state);

  // 3. 重定向到 Google OAuth
  window.location.href = auth_url;
}

// pages/dashboard/adscenter/callback.tsx
export default function OAuthCallback() {
  const router = useRouter();
  const { code, state } = router.query;

  useEffect(() => {
    async function handleCallback() {
      // 1. 验证 state
      const savedState = sessionStorage.getItem('oauth_state');
      if (state !== savedState) {
        toast.error('OAuth 验证失败，请重试');
        return;
      }

      // 2. 发送 code 到后端
      const response = await fetch(`${API_BASE_URL}/adscenter/oauth/callback`, {
        method: 'POST',
        body: JSON.stringify({ code, state }),
      });

      const result = await response.json();

      // 3. 显示成功消息
      toast.success(`成功导入 ${result.accounts_imported} 个 Ads 账号`);

      // 4. 跳转到 Ads 中心
      router.push('/dashboard/adscenter');
    }

    if (code && state) {
      handleCallback();
    }
  }, [code, state]);

  return <LoadingOverlay />;
}
```

## 🔄 Firestore 自动同步配置

### 架构说明

**数据流**:
```
PostgreSQL → Cloud Functions (触发器) → Firestore → 前端 (Realtime Listener)
```

### 1. PostgreSQL 触发器配置

在 PostgreSQL 中创建触发器，当数据变更时发布到 Pub/Sub：

```sql
-- 创建 Pub/Sub 通知函数
CREATE OR REPLACE FUNCTION notify_firestore_sync()
RETURNS TRIGGER AS $$
DECLARE
  payload JSON;
BEGIN
  -- 构建 payload
  payload = json_build_object(
    'table', TG_TABLE_NAME,
    'operation', TG_OP,
    'id', NEW.id,
    'data', row_to_json(NEW)
  );

  -- 发布到 Pub/Sub (通过 pg_notify)
  PERFORM pg_notify('firestore_sync', payload::text);

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 为 offers 表创建触发器
CREATE TRIGGER offers_firestore_sync_trigger
AFTER INSERT OR UPDATE ON offers
FOR EACH ROW
EXECUTE FUNCTION notify_firestore_sync();

-- 为 ads_accounts 表创建触发器
CREATE TRIGGER ads_accounts_firestore_sync_trigger
AFTER INSERT OR UPDATE ON ads_accounts
FOR EACH ROW
EXECUTE FUNCTION notify_firestore_sync();

-- 为 click_tasks 表创建触发器
CREATE TRIGGER click_tasks_firestore_sync_trigger
AFTER INSERT OR UPDATE ON click_tasks
FOR EACH ROW
EXECUTE FUNCTION notify_firestore_sync();
```

### 2. Cloud Function 同步服务

创建 Cloud Function 监听 Pub/Sub 并同步到 Firestore：

```go
// functions/firestore-sync/main.go
package firestoresync

import (
    "context"
    "encoding/json"
    "log"

    "cloud.google.com/go/firestore"
)

type SyncPayload struct {
    Table     string                 `json:"table"`
    Operation string                 `json:"operation"`
    ID        string                 `json:"id"`
    Data      map[string]interface{} `json:"data"`
}

func SyncToFirestore(ctx context.Context, m PubSubMessage) error {
    var payload SyncPayload
    if err := json.Unmarshal(m.Data, &payload); err != nil {
        return err
    }

    client, err := firestore.NewClient(ctx, "your-gcp-project-id")
    if err != nil {
        return err
    }
    defer client.Close()

    // 根据表名决定 Firestore 路径
    var collectionPath string
    switch payload.Table {
    case "offers":
        collectionPath = "offers/" + payload.ID + "/cache"
    case "ads_accounts":
        collectionPath = "ads_accounts/" + payload.ID + "/cache"
    case "click_tasks":
        collectionPath = "click_tasks/" + payload.ID + "/cache"
    default:
        log.Printf("Unknown table: %s", payload.Table)
        return nil
    }

    // 写入 Firestore
    _, err = client.Doc(collectionPath).Set(ctx, payload.Data)
    if err != nil {
        return err
    }

    log.Printf("Synced %s %s to Firestore", payload.Table, payload.ID)
    return nil
}
```

**部署 Cloud Function**:
```bash
gcloud functions deploy firestore-sync \
  --gen2 \
  --runtime=go121 \
  --region=asia-northeast1 \
  --source=./functions/firestore-sync \
  --entry-point=SyncToFirestore \
  --trigger-topic=firestore-sync-topic \
  --service-account=service-account@your-gcp-project-id.iam.gserviceaccount.com
```

### 3. 前端 Realtime Listener

```typescript
// lib/firestore/useOfferCache.ts
import { doc, onSnapshot } from 'firebase/firestore';
import { useEffect, useState } from 'react';

export function useOfferCache(offerId: string) {
  const [offer, setOffer] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!offerId) return;

    const db = getFirestore();
    const offerRef = doc(db, `offers/${offerId}/cache`);

    // 实时监听
    const unsubscribe = onSnapshot(offerRef, (snapshot) => {
      if (snapshot.exists()) {
        setOffer(snapshot.data());
      }
      setLoading(false);
    });

    return () => unsubscribe();
  }, [offerId]);

  return { offer, loading };
}
```

### 4. Firestore 安全规则

```javascript
// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 用户只能读取自己的数据缓存
    match /offers/{offerId}/cache {
      allow read: if request.auth != null &&
                     resource.data.user_id == request.auth.uid;
    }

    match /ads_accounts/{accountId}/cache {
      allow read: if request.auth != null &&
                     resource.data.user_id == request.auth.uid;
    }

    match /click_tasks/{taskId}/cache {
      allow read: if request.auth != null &&
                     resource.data.user_id == request.auth.uid;
    }

    // 用户偏好设置可读写
    match /users/{userId}/preferences {
      allow read, write: if request.auth != null &&
                            request.auth.uid == userId;
    }
  }
}
```

